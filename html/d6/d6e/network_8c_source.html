78 float get_current_rate(network net)
79 {
80  int batch_num = get_current_batch(net);
81  int i;
82  float rate;
83  if (batch_num < net.burn_in) return net.learning_rate * pow((float)batch_num / net.burn_in, net.power);
84  switch (net.policy) {
85  case CONSTANT:
86  return net.learning_rate;
87  case STEP:
88  return net.learning_rate * pow(net.scale, batch_num/net.step);
89  case STEPS:
90  rate = net.learning_rate;
91  for(i = 0; i < net.num_steps; ++i){
92  if(net.steps[i] > batch_num) return rate;
93  rate *= net.scales[i];
94  //if(net.steps[i] > batch_num - 1 && net.scales[i] > 1) reset_momentum(net);
95  }
96  return rate;
97  case EXP:
98  return net.learning_rate * pow(net.gamma, batch_num);
99  case POLY:
100  return net.learning_rate * pow(1 - (float)batch_num / net.max_batches, net.power);
101  case RANDOM:
102  return net.learning_rate * pow(rand_uniform(0,1), net.power);
103  case SIG:
104  return net.learning_rate * (1./(1.+exp(net.gamma*(batch_num - net.step))));
105  default:
106  fprintf(stderr, "Policy is weird!\n");

108  }
